## Overview

FlushLa integrates with [PassCreator](https://passcreator.com) to generate Apple Wallet passes for users. These passes:
- Display the user's real-time cleanliness score
- Update automatically when scores change
- Send push notifications for score updates
- Serve as QR codes for toilet access verification

## Environment Variables

Add your PassCreator API key to `.env`:

```bash
PASSKEYAPI=your-passcreator-api-key
```

<Warning>
The PassCreator API key has full access to create and modify passes. Keep it secure and never expose it in client-side code.
</Warning>

## Creating a Pass

When a user registers, FlushLa creates both a Supabase Auth account and an Apple Wallet pass:

```python
import requests
import os

@app.post("/api/create_passkey")
def create_passkey(request: Request, payload: CreatePasskeyRequest):
    # 1. Create Supabase Auth user (no email confirmation)
    auth_user = supabase_admin.auth.admin.create_user({
        "email": payload.emailRecipient,
        "password": "1234",
        "email_confirm": True
    })
    
    user_id = auth_user.user.id
    
    # 2. Insert into public.users table
    supabase_admin.table("users").insert({
        "id": user_id,
        "username": payload.firstname,
        "email": payload.emailRecipient,
        "score": 41,  # Starting score
        "is_admin": False
    }).execute()
    
    # 3. Call PassCreator API to generate the pass
    PASSKEYAPI = os.getenv("PASSKEYAPI")
    headers = {
        "Authorization": PASSKEYAPI,
        "Content-Type": "application/json"
    }
    
    api_payload = {
        "data": {
            "templateId": payload.templateId,
            "emailRecipient": payload.emailRecipient,
            "firstname": payload.firstname
        }
    }
    
    response = requests.post(
        "https://app.passcreator.com/api/v3/pass",
        headers=headers,
        json=api_payload
    )
    
    if response.status_code >= 400:
        return {
            "error": True,
            "status": response.status_code,
            "raw": response.text
        }
    
    # 4. Extract barcode value (pass identifier)
    pass_id = response.json()["data"]["barcodeValue"]
    
    # 5. Store pass identifier in database
    supabase_admin.table("passes").insert({
        "user_id": user_id,
        "pass_identifier": pass_id
    }).execute()
    
    return {
        "success": True,
        "user_id": user_id,
        "message": "User created and pass issued"
    }
```

## Updating Pass Data

When a user's cleanliness score changes after a toilet inspection, the pass is automatically updated:

```python
def update_user_score(user_id: str):
    # 1. Update score in database
    new_cleanliness_score = calculate_new_score(user_id)
    supabase.table("users")
        .update({"score": int(new_cleanliness_score)})
        .eq("id", user_id)
        .execute()
    
    # 2. Get pass identifier for this user
    identifier = supabase.table("passes")
        .select("pass_identifier")
        .eq("user_id", user_id)
        .single()
        .execute()
        .data.get("pass_identifier")
    
    # Remove first 10 characters from identifier
    identifier = identifier[10:]
    
    # 3. Fetch all passes to find matching identifier
    PASSKEYAPI = os.getenv("PASSKEYAPI")
    headers = {
        "Authorization": PASSKEYAPI,
        "Content-Type": "application/json"
    }
    
    url = "https://app.passcreator.com/api/v3/pass"
    response = requests.get(url, headers=headers)
    
    if response.status_code >= 400:
        print(f"Failed to fetch passes: {response.status_code}")
        return {"error": True, "message": "Failed to fetch passes"}
    
    passes = response.json().get("data", [])
    
    # Find the matching pass
    for p in passes:
        if str(p.get("autoGeneratedId")) == str(identifier):
            identifier = p.get("identifier")
            break
    
    # 4. Update pass with new score
    url = f"https://app.passcreator.com/api/v3/pass/{identifier}"
    payload = {
        "data": {
            "storedValue": int(new_cleanliness_score)
        }
    }
    
    response = requests.patch(url, headers=headers, json=payload)
    
    if response.status_code >= 400:
        print(f"Failed to update pass: {response.status_code}")
    else:
        print(f"Successfully updated pass with score {new_cleanliness_score}")
    
    return {"success": True, "new_score": new_cleanliness_score}
```

<Note>
The `storedValue` field in the PassCreator API corresponds to the cleanliness score displayed on the Apple Wallet pass.
</Note>

## QR Code Verification

The pass barcode is used for toilet access verification:

```python
@app.post("/api/verify-access")
def verify_access(
    request: Request,
    qr_code_data: str = Body(...),
    toilet_id: int = Body(default=1)
):
    # Look up pass by QR code value
    res = supabase.table("passes")
        .select("*, users(*)")
        .eq("pass_identifier", qr_code_data)
        .execute()
    
    pass_data = res.data
    
    if not pass_data:
        return {"access": "denied", "reason": "Invalid pass"}
    
    # Get user information
    pass_row = pass_data[0]
    user_data = pass_row.get("users")
    
    # Check user tier vs toilet tier
    user_score = user_data.get("score", 0)
    user_tier = get_user_tier(user_score)  # 1=Bronze, 2=Silver, 3=Gold
    
    toilet = supabase.table("toilets")
        .select("*")
        .eq("id", toilet_id)
        .single()
        .execute()
        .data
    
    toilet_tier = toilet.get("current_tier", 1)
    
    if user_tier < toilet_tier:
        return {
            "access": "denied",
            "reason": f"Insufficient tier. You need {toilet_tier - user_tier} more level(s).",
            "user_tier": user_tier,
            "required_tier": toilet_tier,
            "points_needed": (50 if toilet_tier == 2 else 80) - user_score
        }
    
    # Grant access and log user in
    request.session["user_id"] = user_data["id"]
    
    return {
        "access": "granted",
        "user": {
            "id": user_data["id"],
            "username": user_data["username"],
            "score": user_score,
            "tier": user_tier
        },
        "toilet": {
            "id": toilet["id"],
            "name": toilet["name"],
            "tier": toilet_tier
        }
    }
```

## API Endpoints Used

### Create Pass

**POST** `https://app.passcreator.com/api/v3/pass`

```json
{
  "data": {
    "templateId": "your-template-id",
    "emailRecipient": "user@example.com",
    "firstname": "John"
  }
}
```

### Get All Passes

**GET** `https://app.passcreator.com/api/v3/pass`

Returns array of all passes with `autoGeneratedId` and `identifier` fields.

### Update Pass

**PATCH** `https://app.passcreator.com/api/v3/pass/{identifier}`

```json
{
  "data": {
    "storedValue": 75
  }
}
```

### Send Push Notification

**POST** `https://app.passcreator.com/api/pass/{identifier}/sendpushnotification`

```json
{
  "pushNotificationText": "Your score has been updated to 75."
}
```

## Score Calculation

User scores are updated based on cleanliness inspection results:

```python
def calculate_score_delta(cleanliness_score: int) -> int:
    """Calculate score delta based on cleanliness score"""
    if cleanliness_score >= 90:
        return 5  # Bonus for very clean
    elif cleanliness_score >= 70:
        return 0  # Neutral
    elif cleanliness_score >= 50:
        return -5  # Small penalty
    elif cleanliness_score >= 30:
        return -10  # Medium penalty
    else:
        return -15  # Large penalty for very dirty
```

## Tier System

Users are assigned tiers based on their score:

```python
def get_user_tier(score: int) -> int:
    """Calculate user tier based on score
    Returns: 1 (Bronze), 2 (Silver), or 3 (Gold)
    """
    if score >= 80:
        return 3  # Gold
    elif score >= 50:
        return 2  # Silver
    else:
        return 1  # Bronze

def get_tier_name(tier: int) -> str:
    """Get tier name from tier number"""
    tier_names = {
        1: "Bronze Standard",
        2: "Silver Standard",
        3: "Gold Standard"
    }
    return tier_names.get(tier, "Unknown")
```

## Complete Workflow

1. **User Registration**: Create Supabase user and PassCreator pass
2. **Toilet Inspection**: AI analyzes toilet image and calculates cleanliness score
3. **Score Update**: Database updated with new score and score delta
4. **Pass Sync**: PassCreator API called to update `storedValue`
5. **Push Notification**: User receives notification on their device
6. **Access Control**: QR code scanned at toilet entrance, tier checked

## Error Handling

```python
try:
    response = requests.patch(url, headers=headers, json=payload)
    
    if response.status_code >= 400:
        print(f"PassCreator API error: {response.status_code} - {response.text}")
        return {"error": True, "message": "Failed to update pass"}
    
    return {"success": True}
except Exception as e:
    print(f"Exception during pass update: {e}")
```
